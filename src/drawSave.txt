class GridSquare {
  constructor(x, y, size) {
    this.x = x;
    this.y = y;
    this.size = size;
  }

  draw() {
    gridContext.fillStyle = '#FFF';
    gridContext.fillRect(this.x, this.y, this.size, this.size);
    gridContext.globalCompositeOperation = 'destination-out';

    // Draw main vertical hole line (going across the entire square)
    gridContext.fillRect(this.x + this.size / 2 - 1, this.y, 2, this.size);

    // Draw main horizontal hole line
    gridContext.fillRect(this.x, this.y + this.size / 2 - 1, this.size, 2);

    // Draw hole lines at the edges of the square
    // Left edge
    gridContext.fillRect(this.x - 1, this.y, 2, this.size);

    // Right edge
    gridContext.fillRect(this.x + this.size - 1, this.y, 2, this.size);

    // Top edge
    gridContext.fillRect(this.x, this.y - 1, this.size, 2);

    // Bottom edge
    gridContext.fillRect(this.x, this.y + this.size - 1, this.size, 2);

    // Draw random vertical line holes in the middle of the smaller squares
    const halfSize = this.size / 2;
    const quarterSize = this.size / 4;

    if (random() < 0.5) { // 50% chance to draw each line
      // Top-left smaller square
      gridContext.fillRect(this.x + quarterSize - 1, this.y, 2, quarterSize);
    }

    if (random() < 0.5) {
      // Top-right smaller square
      gridContext.fillRect(this.x + halfSize + quarterSize - 1, this.y, 2, quarterSize);
    }

    if (random() < 0.5) {
      // Bottom-left smaller square
      gridContext.fillRect(this.x + quarterSize - 1, this.y + halfSize, 2, quarterSize);
    }

    if (random() < 0.5) {
      // Bottom-right smaller square
      gridContext.fillRect(this.x + halfSize + quarterSize - 1, this.y + halfSize, 2, quarterSize);
    }

    gridContext.globalCompositeOperation = 'source-over';
  }
}
